# hooks/review_hook.js
import { query } from "@anthropic-ai/claude-agent-sdk";

async function reviewCode(code) {
  const result = await query({
    prompt: `Review this code for:
1. Security vulnerabilities
2. Performance issues
3. Code style violations

Code:
${code}`,
    options: {
      maxTurns: 1,
      model: "haiku"
    }
  });
  return result.response;
}

// Run on every file edit via PostToolUse

CI/CD Pipelines

GitHub Actions with Claude

Integrate Claude Code into your CI/CD pipeline for automated code review, testing, and deployment assistance.

# .github/workflows/claude-ci.yml
name: Claude Code CI

on:
  pull_request:
    types: [opened, synchronize]
  push:
    branches: [main]

jobs:
  code-review:
    runs-on: ubuntu-latest
    if: github.event_name == 'pull_request'
    steps:
      - uses: actions/checkout@v4
        with:
          fetch-depth: 0

      - name: Setup Node.js
        uses: actions/setup-node@v4
        with:
          node-version: '20'

      - name: Install Claude Code
        run: npm install -g @anthropic-ai/claude-code

      - name: Review PR
        env:
          ANTHROPIC_API_KEY: ${{ secrets.ANTHROPIC_API_KEY }}
        run: |
          claude --headless --print \
            --allowedTools "Read,Grep,Glob" \
            -p "Review the changes in this PR. Focus on:
                1. Security vulnerabilities
                2. Performance issues
                3. Code style consistency
                Output as a structured report."

  auto-fix:
    runs-on: ubuntu-latest
    if: github.event_name == 'push'
    steps:
      - uses: actions/checkout@v4

      - name: Install Claude Code
        run: npm install -g @anthropic-ai/claude-code

      - name: Fix Lint Errors
        env:
          ANTHROPIC_API_KEY: ${{ secrets.ANTHROPIC_API_KEY }}
        run: |
          claude --headless --yes \
            -p "Run npm run lint. Fix any errors. Commit the fixes."

Pre-merge Validation

# Validate before merge
  validate:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - name: Claude Validation
        env:
          ANTHROPIC_API_KEY: ${{ secrets.ANTHROPIC_API_KEY }}
        run: |
          result=$(claude --headless --print \
            --allowedTools "Read,Grep,Bash" \
            -p "Check if this code is production-ready:
                1. Run tests (npm test)
                2. Check for TODO/FIXME comments
                3. Verify no console.log statements
                4. Ensure error handling is complete

                Output PASS or FAIL with reasons.")

          if echo "$result" | grep -q "FAIL"; then
            echo "Validation failed:"
            echo "$result"
            exit 1
          fi

Deployment Assistance

Smart Deployment Pipeline

  deploy:
    runs-on: ubuntu-latest
    needs: [code-review, validate]
    steps:
      - name: Generate Changelog
        run: |
          claude --headless --print \
            -p "Generate changelog from commits since last release" \
            > CHANGELOG_ENTRY.md

      - name: Deploy Check
        run: |
          claude --headless --print \
            --allowedTools "Read,Bash" \
            -p "Pre-deployment checklist:
                1. Database migrations needed?
                2. Environment variables changed?
                3. Breaking API changes?
                Output deployment notes."

      - name: Deploy
        run: npm run deploy

Security Scanning

  security-scan:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - name: Claude Security Scan
        env:
          ANTHROPIC_API_KEY: ${{ secrets.ANTHROPIC_API_KEY }}
        run: |
          claude --headless --print \
            --allowedTools "Read,Grep,Glob" \
            -p "Perform security audit:
                1. Check for hardcoded secrets
                2. SQL injection vulnerabilities
                3. XSS vulnerabilities
                4. Insecure dependencies
                5. Authentication issues

                Output findings with severity levels."

Cost Management Tip

Use --model haiku for quick checks and --allowedTools to limit scope. Reserve Opus for complex analysis. This can reduce CI costs by 80%.

Best Practices

CLAUDE.md Best Practices

# CLAUDE.md

## Project Overview
Brief description of what this project does.

## Architecture
- Frontend: React + TypeScript
- Backend: Node.js + Express
- Database: PostgreSQL

## Coding Conventions
- Use functional components with hooks
- Prefer named exports over default exports
- Error handling: Always use try-catch with typed errors
- Testing: Write tests alongside implementation

## Common Patterns
### API Calls
Use the useQuery hook from @tanstack/react-query:
\`\`\`typescript
const { data, isLoading } = useQuery({
  queryKey: ['users'],
  queryFn: fetchUsers
});
\`\`\`

## Known Issues
- Don't modify files in /legacy - scheduled for removal
- The auth module uses an older pattern - don't follow it

## Deployment
- Staging: Auto-deploy on push to develop
- Production: Manual approval required

Hook Security Best Practices

Do

  • - Validate all input from stdin
  • - Use absolute paths in hooks
  • - Handle errors gracefully
  • - Log hook activity for debugging
  • - Test hooks thoroughly before deploy

Don't

  • - Trust file paths blindly
  • - Execute untrusted commands
  • - Store secrets in hook files
  • - Ignore exit codes
  • - Skip path traversal checks

Effective Prompting

Be Specific

# Bad
"Fix the bug"

# Good
"Fix the null pointer exception in src/auth/login.ts:45
that occurs when the user session expires"

Provide Context

# Bad
"Add validation"

# Good
"Add email validation to the signup form in @src/components/SignupForm.tsx
following the pattern used in @src/components/LoginForm.tsx"

Define Output

# Bad
"Analyze the code"

# Good
"Analyze src/api/ for performance issues.
Output as JSON with: { file, line, issue, suggestion, severity }"

Resource Management

Resource Best Practice
Context Window Use /compact regularly. Reference files with @ instead of pasting.
API Costs Use Haiku for simple tasks. Limit subagent maxTurns.
Memory Keep CLAUDE.md focused. Use # to update not accumulate.
Time Use Plan Mode for complex tasks. Parallel subagents for speed.

Pro Tips (Boris Cherny)

Boris Cherny's Advanced Tips

Anthropic Developer Relations의 Boris Cherny가 공유한 고급 팁들입니다. 이 팁들은 Claude Code를 전문가 수준으로 활용하는 데 핵심입니다.

Tip 1: 병렬 실행 환경 구성

여러 터미널에서 Claude를 동시에 실행하여 작업 처리량을 극대화하세요. 각 터미널에서 다른 서브태스크를 진행하면 전체 작업 시간을 크게 단축할 수 있습니다. 

# 터미널 1: 백엔드 API 개발
cd backend && claude "API 엔드포인트 구현"

# 터미널 2: 프론트엔드 개발
cd frontend && claude "React 컴포넌트 개발"

# 터미널 3: 테스트 작성
cd tests && claude "유닛 테스트 작성"

# 터미널 4: 문서화
cd docs && claude "API 문서 생성"

Tip 2: 웹과 로컬 병렬 운영

Claude.ai (웹)와 Claude Code (터미널)를 동시에 활용하세요. 웹에서 리서치/설계를 하면서 터미널에서 구현을 진행합니다.

Claude.ai (웹)

  • - 아키텍처 설계 논의
  • - API 문서 분석
  • - 복잡한 알고리즘 설명
  • - 장기 컨텍스트 유지

Claude Code (터미널)

  • - 실제 코드 구현
  • - 파일 시스템 접근
  • - Git 작업
  • - 빌드 및 테스트

Tip 3: 모델 선택 전략

작업 복잡도에 따라 적절한 모델을 선택하세요. 기본적으로 Sonnet을 사용하고, 복잡한 추론이 필요할 때만 Opus with thinking을 사용합니다.

모델 사용 케이스 비용
Haiku 간단한 수정, 포매팅, 리네이밍 $
Sonnet (기본) 일반적인 개발 작업, 버그 수정 $$
Opus 4.5 복잡한 아키텍처, 어려운 버그 $$$
Opus + ultrathink 매우 복잡한 추론, 설계 결정 $$$$ 
# 모델 지정 방법
claude --model opus-4-5 "복잡한 아키텍처 설계"

# ultrathink 모드 (Opus에서만)
> ultrathink about the best approach for this problem

Tip 6: Plan 모드와 자동 수락 워크플로우

복잡한 작업은 Plan 모드로 시작하고, 검증된 패턴은 --yes로 자동 수락합니다.

추천 워크플로우

  1. 1. Plan 모드 진입: Shift+Tab x2
  2. 2. 계획 검토: Claude의 실행 계획 확인
  3. 3. 계획 승인 후 실행: 한 단계씩 확인
  4. 4. 반복 작업은 자동화: 검증 후 --yes 사용

Tip 8: 서브에이전트 전략적 활용

서브에이전트를 병렬로 실행하여 작업 속도를 높이세요. 각 에이전트에게 독립적인 작업을 할당합니다. 

# 메인 Claude에서 서브에이전트 생성
> "4개의 병렬 에이전트로 다음을 동시에 진행해줘:
   1. 보안 취약점 스캔
   2. 성능 프로파일링
   3. 코드 품질 분석
   4. 테스트 커버리지 확인"

# Claude가 자동으로:
# - Task 도구로 4개 서브에이전트 생성
# - run_in_background: true로 병렬 실행
# - 결과를 수집하여 통합 리포트 생성

Tip 13: 검증 피드백 루프 (가장 중요)

핵심 원칙

Claude가 코드를 작성한 후 반드시 검증하세요. 빌드, 테스트, 타입 체크를 통해 오류를 즉시 발견하고 수정합니다. 

# 검증 피드백 루프 패턴
1. Claude가 코드 작성
2. 즉시 검증 실행: npm run build && npm test
3. 오류 발생 시 Claude에게 피드백
4. 수정 후 다시 검증
5. 모든 검증 통과할 때까지 반복

# Hook으로 자동화
{
  "hooks": {
    "PostToolUse": [{
      "matcher": "Write|Edit",
      "hooks": [{
        "type": "command",
        "command": "npm run typecheck 2>&1 || true"
      }]
    }]
  }
}

Official Plugin: code-simplifier

✨
code-simplifier

복잡한 코드를 자동으로 단순화하는 플러그인입니다. 중복 제거, 리팩토링 제안, 가독성 개선을 수행합니다. 

# 설치
claude plugins add github:anthropics/code-simplifier

# 사용
/simplify src/complex-module.ts    # 특정 파일 단순화
/simplify --all                    # 전체 프로젝트 분석

용어 설명

병렬 실행 (Parallel Execution)

여러 Claude 인스턴스나 서브에이전트를 동시에 실행하여 작업을 분산 처리하는 방식입니다. 독립적인 작업들을 병렬로 진행하면 전체 완료 시간을 크게 단축할 수 있습니다.

ultrathink (Extended Thinking)

Opus 모델에서 활성화되는 확장 사고 모드입니다. Claude가 응답 전에 더 오래, 깊이 생각합니다. 복잡한 아키텍처 설계나 어려운 디버깅에 유용합니다. ultrathink about...으로 활성화합니다.

검증 피드백 루프

코드 작성 → 검증(빌드/테스트) → 피드백 → 수정의 반복 과정입니다. PostToolUse 훅으로 자동화하면 Claude가 작성한 코드의 품질을 실시간으로 보장할 수 있습니다.

Claude Code SDK

@anthropic-ai/claude-agent-sdk 패키지입니다. JavaScript/TypeScript에서 Claude Code를 프로그래밍 방식으로 호출할 수 있습니다. 자동화 스크립트, CI/CD 파이프라인, 커스텀 도구 개발에 사용됩니다.

Plan 모드

Shift+Tab x2로 진입하는 계획 모드입니다. Claude가 먼저 실행 계획을 세우고 승인을 받은 후에 작업을 진행합니다. 복잡한 작업에서 의도치 않은 변경을 방지합니다.
